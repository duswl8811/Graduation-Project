A. 몬스터가 스폰지역 내
B. 플레이어가 스폰지역 내
C. 플레이어가 몬스터 인식 범위 내

A : 랜덤으로 돌아다닌다.
B : C가 아니면 랜덤으로 돌아다닌다.
C : 추격한다.

!A : 스폰지역으로 돌아간다. 
!B : C 였다면 추격을 멈춘다.
!C : 돌아다닌다.

A > B > C

A 	o- B		o-	Co- 추격
				Cx- 랜덤	
			x-	Co- 랜덤
				Cx- 랜덤
	x- 스폰 back.

B를 없애자!

if (A) // & B)
{
	if( !M.is_player_tracking )	
		M.RandomMoving();

	else if ( M.is_player_tracking )
		M.PlayerTracking();
	
}

else
{
	M.is_player_tracking = false;
	M.BacktoSpawn();
}

// 충돌체 처리로 enter / exit 처리 하자
if (C)
{
	M.is_player_tracking = true;
}
else if (!C) <- 한 번만 처리하고 싶음
{
	M.is_player_tracking = false;
}

// 충돌체 이용
void OnTriggerStay(Collider other)
{
	// 트랙킹 리스트를 받고 해당 리스트의 사람이 빠져 나가면 --;

	// 새 플레이어가 충돌되면 검사
	PlayerStatus other_status = other.gameObject.GetComponent<PlayerStatus>();
	if (other_status에서 몬스터 거리 < 기존 타겟과 몬스터 거리)
		target = other_status;
	M.is_player_tracking = true;
}
void OnTriggerExit(Collider other)
{
	M.is_player_tracking = false;

	// 트랙킹 카운터 세고 0 일때만 false?
	target = 없음;
}
